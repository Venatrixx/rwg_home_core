Future<void> fetchAll(HipConfig hipConfig) async {
  _loadingState = LoadingState.loading;

  List<LoadingState> states = [];

  try {
    //hip = await fetchHipData(hipConfig).timeout(timeoutDuration);
    hip?.onDataChanged = onDataChanged;
    states.add(LoadingState.done);
  } catch (e) {
    error = e;
    states.add(LoadingState.error);
  }

  try {
    //schedule = await fetchScheduleData().timeout(timeoutDuration);
    schedule?.onDataChanged = onDataChanged;
    states.add(LoadingState.done);
  } catch (e) {
    error = e;
    states.add(LoadingState.error);
  }

  try {
    //calendar = await fetchCalendarData().timeout(timeoutDuration);
    calendar?.onDataChanged = onDataChanged;
    states.add(LoadingState.done);
  } catch (e) {
    error = e;
    states.add(LoadingState.error);
  }

  if (states.any((e) => e == LoadingState.error)) {
    if (states.every((e) => e == LoadingState.error)) {
      _loadingState = LoadingState.error;
      return;
    } else {
      _loadingState = LoadingState.doneWithError;
      return;
    }
  }

  onDataChanged?.call();

  _loadingState = LoadingState.done;
  return;
}
